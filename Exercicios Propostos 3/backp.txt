algoritmo de busca binária (portanto, é necessário que você ordene o seu vetor
antes de fazer a busca).
*/

#include <stdio.h>
#include <math.h>

int vetor[9];

int buscarx(int x)
{
    int ini=0, fim=9, meio;
    while(ini<=fim){
        meio=(ini+fim)/2;

        if(vetor[meio]==x) return meio;
        if(vetor[meio]<x) ini=meio+1;
        if(vetor[meio]>x) fim=meio-1;
    }
    return -1;
}

int main()
{
    int i = 0, aux = 0, x = 0;

    printf("Preencha o vetor com 10 valores!\n");

    while(i < 10){
        printf("Digite o valor para a posição %d: ", i+1);
        scanf("%d", &vetor[i]);
        i+=1;
    }
    i = 0;
    for(int i = 0; i < 10; i++){
        for(int j = 0; j < 10; j++){
            if(vetor[i] < vetor[j]){
                aux = vetor[i];
                vetor[i] = vetor[j];
                vetor[j] = aux;
            }
        }
    }
    printf("Digite o valor a ser localizado: \n");
    scanf("%d", &x);
    printf("O valor foi encontrado na posição %d do vetor.\n", buscarx(x)+1);

    return 0;
}
